    HOW TO ADD ITEMS:
    ─────────────────
    Each item uses: data-cat="poster|video|web|photo"
                    data-size="sm|md|lg|xl|tall|wide"

    Size classes:
        sm   → 1 col  × 1 row  (small square)
        md   → 1 col  × 1 row  (standard)
        lg   → 2 cols × 1 row  (landscape wide)
        xl   → 2 cols × 2 rows (large feature)
        tall → 1 col  × 2 rows (portrait / vertical poster)
        wide → 3 cols × 1 row  (ultra-wide banner)

    For IMAGES:  set data-type="image"  and put <img src="..."> inside .pf-media
    For VIDEOS:  set data-type="video"  and put <video src="..."> inside .pf-media
                    OR set data-video-url="https://youtube-or-vimeo-embed-url"

    <script>
        (function () {
            /* ── Inject overlays into each card ── */
            document.querySelectorAll('.pf-item').forEach(item => {
                const cat = item.dataset.cat || 'poster';
                const title = item.dataset.title || 'Untitled';
                const desc = item.dataset.desc || '';

                const overlay = document.createElement('div');
                overlay.className = 'pf-overlay';
                overlay.innerHTML = `
                                    <div class="pf-overlay-title">${title}</div>
                                    <div class="pf-overlay-sub">${desc}</div>
                                    <div class="pf-overlay-meta">
                                        <span class="pf-chip ${cat}">${cat}</span>
                                    </div>
                                `;
                item.appendChild(overlay);
            });

            /* ── Filter logic ── */
            const filters = document.querySelectorAll('.pf-filter');
            const items = document.querySelectorAll('.pf-item');

            filters.forEach(btn => {
                btn.addEventListener('click', () => {
                    filters.forEach(b => b.classList.remove('active'));
                    btn.classList.add('active');
                    const f = btn.dataset.filter;
                    items.forEach(item => {
                        const show = f === 'all' || item.dataset.cat === f;
                        item.classList.toggle('hidden', !show);
                    });
                });
            });

            /* ── Video hover play/pause ── */
            items.forEach(item => {
                const vid = item.querySelector('video');
                if (!vid) return;
                item.addEventListener('mouseenter', () => vid.play().catch(() => { }));
                item.addEventListener('mouseleave', () => { vid.pause(); vid.currentTime = 0; });
            });

            /* ── Lightbox ── */
            const lb = document.getElementById('pf-lightbox');
            const lbMedia = document.getElementById('lb-media');
            const lbTitle = document.getElementById('lb-title');
            const lbDesc = document.getElementById('lb-desc');
            const lbClose = document.getElementById('lb-close');
            const lbBack = document.getElementById('lb-backdrop');
            const lbPrev = document.getElementById('lb-prev');
            const lbNext = document.getElementById('lb-next');

            let visibleItems = [];
            let currentIdx = 0;

            function getVisible() {
                return [...document.querySelectorAll('.pf-item:not(.hidden)')];
            }

            function openLightbox(idx) {
                visibleItems = getVisible();
                currentIdx = idx;
                renderLb();
                lb.classList.add('open');
                document.body.style.overflow = 'hidden';
            }

            function closeLightbox() {
                lb.classList.remove('open');
                document.body.style.overflow = '';
                lbMedia.innerHTML = '';
            }

            function renderLb() {
                const item = visibleItems[currentIdx];
                const type = item.dataset.type || 'image';
                const title = item.dataset.title || 'Untitled';
                const desc = item.dataset.desc || '';
                const vurl = item.dataset.videoUrl || '';

                lbTitle.textContent = title;
                lbDesc.textContent = desc;
                lbMedia.innerHTML = '';

                if (type === 'embed' && vurl) {
                    const iframe = document.createElement('iframe');
                    iframe.src = vurl + '?autoplay=1';
                    iframe.allow = 'autoplay; fullscreen';
                    lbMedia.appendChild(iframe);
                } else if (type === 'video') {
                    const vid = item.querySelector('video');
                    if (vid) {
                        const v2 = document.createElement('video');
                        v2.src = vid.src;
                        v2.controls = true;
                        v2.autoplay = true;
                        v2.style.maxWidth = '100%';
                        v2.style.maxHeight = '75vh';
                        lbMedia.appendChild(v2);
                    }
                } else {
                    const img = item.querySelector('img');
                    if (img) {
                        const i2 = document.createElement('img');
                        i2.src = img.src;
                        i2.alt = title;
                        lbMedia.appendChild(i2);
                    }
                }

                lbPrev.style.display = currentIdx === 0 ? 'none' : 'flex';
                lbNext.style.display = currentIdx === visibleItems.length - 1 ? 'none' : 'flex';
            }

            items.forEach((item, i) => {
                item.addEventListener('click', () => {
                    const vis = getVisible();
                    const visIdx = vis.indexOf(item);
                    if (visIdx !== -1) openLightbox(visIdx);
                });
            });

            lbClose.addEventListener('click', closeLightbox);
            lbBack.addEventListener('click', closeLightbox);
            lbPrev.addEventListener('click', () => { if (currentIdx > 0) { currentIdx--; renderLb(); } });
            lbNext.addEventListener('click', () => { if (currentIdx < visibleItems.length - 1) { currentIdx++; renderLb(); } });

            document.addEventListener('keydown', e => {
                if (!lb.classList.contains('open')) return;
                if (e.key === 'Escape') closeLightbox();
                if (e.key === 'ArrowLeft') { if (currentIdx > 0) { currentIdx--; renderLb(); } }
                if (e.key === 'ArrowRight') { if (currentIdx < visibleItems.length - 1) { currentIdx++; renderLb(); } }
            });
        })();
    </script>
