        (function () {
            'use strict';

            /* ══════════════════════════════════════════════════
                1. INJECT OVERLAYS + DURATION BADGES
            ══════════════════════════════════════════════════ */
            document.querySelectorAll('.pf-item').forEach(item => {
                const cat = item.dataset.cat || 'poster';
                const type = item.dataset.type || 'image';
                const dur = item.dataset.duration;

                /* overlay */
                const ov = document.createElement('div');
                ov.className = 'pf-overlay';
                ov.innerHTML =
                    '<div class="pf-ov-title">' + (item.dataset.title || 'Untitled') + '</div>' +
                    '<div class="pf-ov-desc">' + (item.dataset.desc || '') + '</div>' +
                    '<span class="pf-chip ' + cat + '">' + cat + (type !== 'image' ? ' · ' + type : '') + '</span>';
                item.appendChild(ov);

                /* duration badge for video items (if not already in HTML) */
                if (dur && !item.querySelector('.pf-dur')) {
                    const d = document.createElement('div');
                    d.className = 'pf-dur';
                    d.textContent = dur;
                    item.appendChild(d);
                }
            });

            /* ══════════════════════════════════════════════════
                2. LAZY LOADING — IntersectionObserver
            ══════════════════════════════════════════════════ */
            function loadMedia(el) {
                const src = el.dataset.src;
                if (!src) return;

                const skel = el.closest('.pf-media')?.querySelector('.pf-skeleton');
                const errEl = el.closest('.pf-media')?.querySelector('.pf-error');

                if (el.tagName === 'IMG') {
                    el.onload = () => {
                        el.classList.add('loaded');
                        if (skel) skel.classList.add('done');
                    };
                    el.onerror = () => {
                        if (skel) skel.classList.add('done');
                        if (errEl) { errEl.classList.add('show'); }
                        else {
                            const e = buildErrorEl('Image failed to load', src);
                            el.closest('.pf-media').appendChild(e);
                        }
                    };
                    el.src = src;
                    delete el.dataset.src;

                } else if (el.tagName === 'VIDEO') {
                    el.onerror = () => {
                        if (skel) skel.classList.add('done');
                        const e = buildErrorEl('Video failed to load', src);
                        el.closest('.pf-media').appendChild(e);
                    };
                    el.src = src;
                    el.load();
                    delete el.dataset.src;
                    if (skel) skel.classList.add('done');
                }
            }

            function buildErrorEl(msg, hint) {
                const e = document.createElement('div');
                e.className = 'pf-error show';
                e.innerHTML =
                    '<div class="pf-error-icon"><svg viewBox="0 0 24 24" fill="none" stroke="#ff5060" stroke-width="1.5"><circle cx="12" cy="12" r="10"/><line x1="15" y1="9" x2="9" y2="15"/><line x1="9" y1="9" x2="15" y2="15"/></svg></div>' +
                    '<span>' + msg + '</span>' +
                    '<small>' + hint + '</small>';
                return e;
            }

            const lazyObs = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (!entry.isIntersecting) return;
                    const el = entry.target;
                    loadMedia(el);
                    lazyObs.unobserve(el);
                });
            }, { rootMargin: '200px' });

            document.querySelectorAll('img[data-src], video[data-src], .pf-thumb[data-src]').forEach(el => {
                lazyObs.observe(el);
            });

            /* ══════════════════════════════════════════════════
                3. VIDEO HOVER — safe play/pause with Promise guard
            ══════════════════════════════════════════════════ */
            document.querySelectorAll('.pf-item').forEach(item => {
                const vid = item.querySelector('video');
                if (!vid) return;

                let playPromise = null;

                item.addEventListener('mouseenter', () => {
                    /* load if not yet loaded */
                    if (vid.dataset.src) {
                        vid.src = vid.dataset.src;
                        vid.load();
                        delete vid.dataset.src;
                        const skel = item.querySelector('.pf-skeleton');
                        if (skel) skel.classList.add('done');
                    }
                    playPromise = vid.play();
                    if (playPromise !== undefined) {
                        playPromise.catch(() => { }); /* silence AbortError */
                    }
                });

                item.addEventListener('mouseleave', () => {
                    if (playPromise !== undefined) {
                        playPromise.then(() => {
                            vid.pause();
                            vid.currentTime = 0;
                        }).catch(() => {
                            vid.pause();
                            vid.currentTime = 0;
                        });
                    } else {
                        vid.pause();
                        vid.currentTime = 0;
                    }
                    playPromise = null;
                });
            });

            /* ══════════════════════════════════════════════════
                4. FILTER
            ══════════════════════════════════════════════════ */
            const filters = document.querySelectorAll('.pf-filter');
            const allItems = [...document.querySelectorAll('.pf-item')];
            const countEl = document.getElementById('pf-count');

            function updateCount() {
                const vis = allItems.filter(i => !i.classList.contains('pf-hidden')).length;
                countEl.textContent = vis + ' item' + (vis !== 1 ? 's' : '');
            }

            filters.forEach(btn => {
                btn.addEventListener('click', () => {
                    filters.forEach(b => b.classList.remove('active'));
                    btn.classList.add('active');
                    const f = btn.dataset.filter;
                    allItems.forEach((item, i) => {
                        const show = f === 'all' || item.dataset.cat === f;
                        item.classList.toggle('pf-hidden', !show);
                        if (show) {
                            item.style.animationDelay = (i * 0.04) + 's';
                            item.classList.remove('pf-animating');
                            void item.offsetWidth;
                            item.classList.add('pf-animating');
                        }
                    });
                    updateCount();
                });
            });
            updateCount();

            /* ══════════════════════════════════════════════════
                5. LIGHTBOX
            ══════════════════════════════════════════════════ */
            const lb = document.getElementById('pf-lb');
            const lbStage = document.getElementById('lb-stage');
            const lbSpin = document.getElementById('lb-spin');
            const lbErrEl = document.getElementById('lb-err');
            const lbTitle = document.getElementById('lb-title');
            const lbDesc = document.getElementById('lb-desc');
            const lbCounter = document.getElementById('lb-counter');
            const lbClose = document.getElementById('lb-close');
            const lbBg = document.getElementById('lb-bg');
            const lbPrev = document.getElementById('lb-prev');
            const lbNext = document.getElementById('lb-next');
            const lbFs = document.getElementById('lb-fs');

            let activeItems = [];
            let curIdx = 0;
            let lbMediaEl = null; /* current media element inside stage */

            /* ── get currently visible items ── */
            function getVisible() { return allItems.filter(i => !i.classList.contains('pf-hidden')); }

            /* ── open ── */
            function openLb(idx) {
                activeItems = getVisible();
                curIdx = Math.max(0, Math.min(idx, activeItems.length - 1));
                lb.classList.add('open');
                document.body.style.overflow = 'hidden';
                renderLb();
            }

            /* ── close ── */
            function closeLb() {
                lb.classList.remove('open');
                document.body.style.overflow = '';
                clearStage();
            }

            /* ── clear stage ── */
            function clearStage() {
                if (lbMediaEl) {
                    if (lbMediaEl.tagName === 'VIDEO') {
                        lbMediaEl.pause();
                        lbMediaEl.src = '';
                        lbMediaEl.load();
                    }
                    if (lbMediaEl.tagName === 'IFRAME') {
                        lbMediaEl.src = '';
                    }
                    lbMediaEl.remove();
                    lbMediaEl = null;
                }
                /* remove any old media */
                [...lbStage.querySelectorAll('img,video,iframe')].forEach(e => e.remove());
                lbSpin.classList.remove('done');
                lbErrEl.classList.remove('show');
            }

            /* ── render ── */
            function renderLb() {
                clearStage();

                const item = activeItems[curIdx];
                const type = item.dataset.type || 'image';
                const title = item.dataset.title || 'Untitled';
                const desc = item.dataset.desc || '';
                const vid_id = item.dataset.videoId || '';

                lbTitle.textContent = title;
                lbDesc.textContent = desc;
                lbCounter.textContent = (curIdx + 1) + ' / ' + activeItems.length;
                lbPrev.disabled = curIdx === 0;
                lbNext.disabled = curIdx === activeItems.length - 1;

                /* ── IMAGE ── */
                if (type === 'image') {
                    const srcEl = item.querySelector('img[src], img[data-src]');
                    const src = srcEl ? (srcEl.src || srcEl.dataset.src) : null;

                    if (!src) { showLbError('No image source found'); return; }

                    const img = document.createElement('img');
                    img.alt = title;
                    img.style.opacity = '0';
                    img.style.transition = 'opacity .3s';
                    lbStage.appendChild(img);
                    lbMediaEl = img;

                    img.onload = () => {
                        lbSpin.classList.add('done');
                        img.style.opacity = '1';
                    };
                    img.onerror = () => showLbError('Image failed to load');
                    img.src = src;

                    /* ── LOCAL VIDEO ── */
                } else if (type === 'video') {
                    const srcEl = item.querySelector('video');
                    const src = srcEl ? (srcEl.src || srcEl.dataset.src || srcEl.currentSrc) : null;
                    const poster = srcEl ? srcEl.poster : null;

                    if (!src) { showLbError('No video source found'); return; }

                    const video = document.createElement('video');
                    video.controls = true;
                    video.autoplay = true;
                    video.playsInline = true;
                    if (poster) video.poster = poster;
                    lbStage.appendChild(video);
                    lbMediaEl = video;

                    video.oncanplay = () => { lbSpin.classList.add('done'); };
                    video.onerror = () => showLbError('Video failed to load');

                    /* Source element for better format handling */
                    const s = document.createElement('source');
                    s.src = src;
                    s.type = src.endsWith('.webm') ? 'video/webm' :
                        src.endsWith('.ogv') ? 'video/ogg' : 'video/mp4';
                    video.appendChild(s);
                    video.load();

                    /* ── YOUTUBE ── */
                } else if (type === 'youtube') {
                    if (!vid_id) { showLbError('No YouTube video ID'); return; }

                    const iframe = document.createElement('iframe');
                    const params = 'autoplay=1&rel=0&modestbranding=1&playsinline=1';
                    iframe.src = 'https://www.youtube.com/embed/' + vid_id + '?' + params;
                    iframe.allow = 'autoplay; fullscreen; picture-in-picture';
                    iframe.allowFullscreen = true;
                    iframe.style.aspectRatio = '16/9';
                    iframe.style.width = '100%';

                    iframe.onload = () => lbSpin.classList.add('done');
                    iframe.onerror = () => showLbError('YouTube failed to load');

                    lbStage.appendChild(iframe);
                    lbMediaEl = iframe;
                    /* YouTube iframe doesn't fire onerror reliably, fallback timer */
                    setTimeout(() => lbSpin.classList.add('done'), 2500);

                    /* ── VIMEO ── */
                } else if (type === 'vimeo') {
                    if (!vid_id) { showLbError('No Vimeo video ID'); return; }

                    const iframe = document.createElement('iframe');
                    const params = 'autoplay=1&title=0&byline=0&portrait=0';
                    iframe.src = 'https://player.vimeo.com/video/' + vid_id + '?' + params;
                    iframe.allow = 'autoplay; fullscreen; picture-in-picture';
                    iframe.allowFullscreen = true;
                    iframe.style.aspectRatio = '16/9';
                    iframe.style.width = '100%';

                    iframe.onload = () => lbSpin.classList.add('done');
                    lbStage.appendChild(iframe);
                    lbMediaEl = iframe;
                    setTimeout(() => lbSpin.classList.add('done'), 2500);
                }
            }

            function showLbError(msg) {
                lbSpin.classList.add('done');
                lbErrEl.querySelector('p').textContent = msg;
                lbErrEl.classList.add('show');
            }

            /* ── Fullscreen ── */
            lbFs.addEventListener('click', () => {
                if (!document.fullscreenElement) {
                    lbStage.requestFullscreen().catch(() => { });
                } else {
                    document.exitFullscreen().catch(() => { });
                }
            });

            /* ── Navigation ── */
            lbPrev.addEventListener('click', () => { if (curIdx > 0) { curIdx--; renderLb(); } });
            lbNext.addEventListener('click', () => { if (curIdx < activeItems.length - 1) { curIdx++; renderLb(); } });
            lbClose.addEventListener('click', closeLb);
            lbBg.addEventListener('click', closeLb);

            document.addEventListener('keydown', e => {
                if (!lb.classList.contains('open')) return;
                if (e.key === 'Escape') closeLb();
                if (e.key === 'ArrowLeft') { if (curIdx > 0) { curIdx--; renderLb(); } }
                if (e.key === 'ArrowRight') { if (curIdx < activeItems.length - 1) { curIdx++; renderLb(); } }
            });

            /* ── Click items to open ── */
            allItems.forEach(item => {
                item.addEventListener('click', () => {
                    const vis = getVisible();
                    const idx = vis.indexOf(item);
                    if (idx !== -1) openLb(idx);
                });
            });

            /* ══════════════════════════════════════════════════
                6. TOUCH SWIPE IN LIGHTBOX
            ══════════════════════════════════════════════════ */
            let touchStartX = 0;
            const lbWrap = document.getElementById('lb-wrap');
            lbWrap.addEventListener('touchstart', e => { touchStartX = e.changedTouches[0].screenX; }, { passive: true });
            lbWrap.addEventListener('touchend', e => {
                const dx = e.changedTouches[0].screenX - touchStartX;
                if (Math.abs(dx) < 40) return;
                if (dx < 0 && curIdx < activeItems.length - 1) { curIdx++; renderLb(); }
                if (dx > 0 && curIdx > 0) { curIdx--; renderLb(); }
            });

        })();
